Arduino LED circuits


Button toggled LED

// C++ code
//
const int readButton = 4;
const int led = 8;
int buttonState, toggleAllow = 0, ledState = 0;

void setup()
{
pinMode(LED_BUILTIN, OUTPUT);
pinMode(readButton, INPUT);
pinMode(led, OUTPUT);
}

void loop()
{
digitalWrite(LED_BUILTIN, HIGH);

buttonState = digitalRead(readButton);

if (toggleAllow == 0 && buttonState == 1)
{
if (ledState == 0)
{
digitalWrite(led, 1);
ledState = 1;
}
else
{
digitalWrite(led, 0);
ledState = 0;
}
}

toggleAllow = buttonState;
}
